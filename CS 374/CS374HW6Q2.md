

```pseudocode
get i word =
	word ?= "Ring" -> A[i]
	word ?= "Ding" -> -A[i]
	
negate word 

maxChicken n word count:
	count ?= 3:
		get i (negate word)
	otherwise:
		max(get i word, get i (negate word))

maxChicken i word count:
	count ?= 3:
		<~ get i (negate word) + maxChicken (i+1) (negate word) 1
	otherwise:
		<~ max(
			get i word + maxChicken (i+1) word (count + 1),
			get i (negate word) + maxChicken (i+1) (negate word) 1
		   )
			
return maxChicken 1 ("Ring" or "Ding") 0
```
